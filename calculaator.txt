ans="Y"
loop do
puts "Enter first number"
fnum=gets.to_i

puts "Enter second number"
snum=gets.to_i

puts "Enter operator"
op = gets.chomp

if op== "+"
result= fnum*snum
puts result

elsif op== "-"
result= fnum*snum
puts result

elsif op== "*"
result= fnum*snum
puts result

elsif op== "/"
result= fnum/snum
puts result

else 
puts "error"
end

File.open("step3.txt","a") do |line|
line.puts "\r" +fnum.to_s 
line.puts "\r" +snum.to_s 
line.puts "\r" + op.to_s
line.puts "\r" + result.to_s
end

puts "Do another Y/N"
ans=gets.chomp
break if(ans=="N")
end
